import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int k = scanner.nextInt();
        String S = scanner.next();
        scanner.close();

        Map<Character, Integer> d = new HashMap<>();
        for (char c : S.toCharArray()) {
            try {
                d.put(c, d.get(c) + 1);
            } catch (Exception e) {
                d.put(c, 1);
            }
        }

        boolean done = false;
        for (Map.Entry<Character, Integer> entry : d.entrySet()) {
            if (entry.getValue() % k != 0) {
                System.out.println(-1);
                done = true;
                break;
            }
        }

        if (!done) {
            String o = "";
            for (Map.Entry<Character, Integer> entry : d.entrySet()) {
                o += entry.getKey() + (entry.getValue() / k);
            }
            System.out.println(o * k);
        }
    }
}

  [JAVA]
import java.util.Scanner;
import java.util.HashMap;

public class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int k = scanner.nextInt();
    String s = scanner.next();
    HashMap<Character, Integer> d = new HashMap<>();
    for (int i = 0; i < s.length(); i++) {
      char c = s.charAt(i);
      if (d.containsKey(c)) {
        d.put(c, d.get(c) + 1);
      } else {
        d.put(c, 1);
      }
    }
    boolean done = false;
    for (Character c : d.keySet()) {
      if (d.get(c) % k != 0) {
        System.out.println(-1);
        done = true;
        break;
      }
    }
    if (!done) {
      StringBuilder sb = new StringBuilder();
      for (Character c : d.keySet()) {
        sb.append(c.toString().repeat(d.get(c) / k));
      }
      System.out.println(sb.toString().repeat(k));
    }
  }
}

import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int k = scanner.nextInt();
        scanner.nextLine();
        String s = scanner.nextLine();
        scanner.close();

        boolean done = false;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == ' ') {
                continue;
            }
            if (c < 'a' || c > 'z') {
                System.out.println(-1);
                done = true;
                break;
            }
            if (c - 'a' + 1 % k != 0) {
                System.out.println(-1);
                done = true;
                break;
            }
        }

        if (!done) {
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < s.length(); i++) {
                char c = s.charAt(i);
                if (c == ' ') {
                    continue;
                }
                sb.append(c);
            }
            String result = sb.toString();
            System.out.println(result.repeat(k));
        }
    }
}

        int k = scanner.nextInt();

        HashMap<Character, Integer> d = new HashMap<>();

                d.put(s, d.get(s) + 1);
            } catch (NullPointerException e) {
                d.put(s, 1);
            }
        }
        boolean done = false;
        for (char x : d.keySet()) {
            if (d.get(x) % k != 0) {
                System.out.println(-1);
                done = true;
                break;
            }
        }
        if (!done) {
            String o = "";
            for (char x : d.keySet()) {
                o += String.valueOf(x).repeat(d.get(x) / k);
            }
            System.out.println(o.repeat(k));
        }

public class Solution {

        int k = scanner.nextInt();
        String S = scanner.next();
        scanner.close();

        Map<Character, Integer> d = new HashMap<>();
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (d.containsKey(c)) {
                d.put(c, d.get(c) + 1);
            } else {
                d.put(c, 1);
            }
        }

        boolean done = false;

                char c = entry.getKey();
                int count = entry.getValue() / k;
                for (int i = 0; i < count; i++) {
                    sb.append(c);
                }
            }
            System.out.println(sb.toString() * k);
        }
    }
