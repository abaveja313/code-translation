import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int lineCounter = 0;
        int whiteCounter = 0;
        int blackCounter = 0;

        while (lineCounter != 8) {
            String line = input.nextLine();
            lineCounter++;
            for (int i = 0; i < line.length(); i++) {
                char ch = line.charAt(i);
                if (ch == 'Q') {
                    whiteCounter += 9;
                } else if (ch == 'R') {
                    whiteCounter += 5;
                } else if (ch == 'B') {
                    whiteCounter += 3;
                } else if (ch == 'N') {
                    whiteCounter += 3;
                } else if (ch == 'P') {
                    whiteCounter += 1;
                } else if (ch == 'q') {
                    blackCounter += 9;
                } else if (ch == 'r') {
                    blackCounter += 5;
                } else if (ch == 'b') {
                    blackCounter += 3;
                } else if (ch == 'n') {
                    blackCounter += 3;
                } else if (ch == 'p') {
                    blackCounter += 1;
                }
            }
        }

        if (whiteCounter > blackCounter) {
            System.out.println("White");
        } else if (blackCounter > whiteCounter) {
            System.out.println("Black");
        } else {
            System.out.println("Draw");
        }
    }
}
