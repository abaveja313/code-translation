        Scanner scanner = new Scanner(System.in);
        String ans = "YES";
        for (int i = 0; i < 8; i++) {
            String line = scanner.nextLine();
            for (int j = 0; j < 7; j++) {
                if (line.charAt(j) == line.charAt(j + 1)) {
                    ans = "NO";
                    break;
                }
            }
        }
        System.out.println(ans);
    }
}

The `break` statement is also used in the same way as in Python. The `charAt` method is used to access the characters in the string.

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String ans = "YES";
        for (int i = 0; i < 8; i++) {
            String line = sc.nextLine();
            for (int j = 0; j < 7; j++) {
                if (line.charAt(j) == line.charAt(j + 1)) {
                    ans = "NO";
                    break;
                }
            }
        }
        System.out.println(ans);
    }
}
