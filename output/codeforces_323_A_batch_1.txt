import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int k = scanner.nextInt();
        if (k % 2 == 1) {
            System.out.println(-1);
        } else {
            for (int i = 0; i < k; i++) {
                for (int j = 0; j < k; j++) {
                    String a = "";
                    for (int l = 0; l < k; l++) {
                        a += ((i / 2 + j + l / 2) % 2 == 0) ? "b" : "w";
                    }
                    System.out.println(a);
                }
                if (i < k - 1) {
                    System.out.println();
                }
            }
        }
    }
}

    Scanner scanner = new Scanner(System.in);
    int k = scanner.nextInt();
    if (k % 2 == 1) {
      System.out.println(-1);
    } else {
      for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
          String a = "";
          for (int l = 0; l < k; l++) {
            a += (i / 2 + j + l / 2) % 2 == 0 ? "b" : "w";
          }
          System.out.print(a);
        }
        if (i < k - 1) {
          System.out.println();
        }
      }
    }
  }
}
