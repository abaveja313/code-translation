    }

    private static int findSmallest(int[] array) {
        int smallest = array[0];
        for (int i = 1; i < array.length; i++) {
            if (array[i] < smallest) {
                smallest = array[i];
            }
        }

import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String[] array = sc.nextLine().split(" ");
        int[] intArray = new int[n];
        for (int i = 0; i < n; i++) {
            intArray[i] = Integer.parseInt(array[i]);
        }
        int smallest = getSmallest(intArray);
        if (isDivisible(intArray, smallest)) {
            System.out.println(smallest);
        } else {
            System.out.println(-1);
        }
    }

    private static int getSmallest(int[] array) {
        int smallest = array[0];
        for (int i = 1; i < array.length; i++) {
            if (array[i] < smallest) {
                smallest = array[i];
            }
        }
        return smallest;
    }

    private static boolean isDivisible(int[] array, int smallest) {
        for (int i = 1; i < array.length; i++) {
            if (array[i] % smallest != 0) {
                return false;
            }
        }
        return true;
    }
}
