        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int[] AN = new int[N];
        for (int i = 0; i < N; i++) {
            AN[i] = sc.nextInt();

            int length = 0;

                    digitsCt[length + 1] = one;
                } else {
                    digitsCt[length] = zero;
                    digitsCt[length + 1] = (one + 1) % MOD;
                }
                temp = temp >> 1;
                length++;
            }
        }

            int one = digitsCt[i + 1];
