import java.util.Scanner;

class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int n = scanner.nextInt();
    int[] s = new int[n];
    for (int i = 0; i < n; i++) {
      s[i] = scanner.nextInt();
    }
    int max = s[0];
    for (int i = 1; i < n; i++) {
      if (s[i] > max) {
        max = s[i];
      }
    }
    System.out.println(max);
  }
}
