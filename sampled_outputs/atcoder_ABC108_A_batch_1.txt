import java.util.Scanner;

class Solution {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int K = sc.nextInt();
    int odd = (K + 1) / 2;
    int even = K - odd;
    System.out.println(odd * even);
  }
}
