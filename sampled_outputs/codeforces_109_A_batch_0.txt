import java.util.Scanner;

class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int s = scanner.nextInt();
    int n = 0, m = 0;
    for (int i = 0; i <= s / 4; i++) {
      if ((s - 4 * i) % 7 == 0) {
        n = i;
        m = (s - 4 * i) / 7;
        break;
      }
    }
    if (n == 0 && m == 0) {
      System.out.println(-1);
    } else {
      System.out.println("4".repeat(n) + "7".repeat(m));
    }
  }
}
