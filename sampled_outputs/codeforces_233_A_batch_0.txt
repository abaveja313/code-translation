import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();
        if (a % 2 == 1) {
            System.out.println("-1");
        } else {
            for (int i = 1; i <= a; i++) {
                if (i % 2 == 1) {
                    System.out.print(i + 1 + " ");
                } else {
                    System.out.print(i - 1 + " ");
                }
            }
        }
    }
}
