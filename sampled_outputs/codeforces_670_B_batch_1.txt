import java.util.Scanner;

class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int n = scanner.nextInt();
    int k = scanner.nextInt();
    String[] s = scanner.nextLine().split(" ");

    for (int i = 1; i <= n; i++) {
      if (k - i > 0) {
        k -= i;
      } else {
        System.out.println(s[k - 1]);
        break;
      }
    }
  }
}
