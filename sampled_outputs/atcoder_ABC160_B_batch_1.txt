import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int x = scanner.nextInt();
        int a = x / 500;
        int b = x % 500;
        int c = b / 5;
        int ans = 1000 * a + 5 * c;
        System.out.println(ans);
    }
}
