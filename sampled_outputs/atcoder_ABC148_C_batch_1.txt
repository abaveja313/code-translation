import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
        int min = Math.min(a, b);
        int ret = a * b;
        for (int i = min; i > 0; i--) {
            if (a % i == 0 && b % i == 0) {
                ret = a * b / i;
                break;
            }
        }
        System.out.println(ret);
    }
}
