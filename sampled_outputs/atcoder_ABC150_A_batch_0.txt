class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int k = scanner.nextInt();
        int x = scanner.nextInt();
        if (k * 500 >= x) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
