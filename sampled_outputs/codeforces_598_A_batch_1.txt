import java.util.Scanner;

public class Solution {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int testCase = sc.nextInt();

    while (testCase > 0) {
      int n = sc.nextInt();
      int sum = n * (n + 1) / 2;
      int i = 1;

      while (i <= n) {
        sum -= i * 2;
        i *= 2;
      }

      System.out.println(sum);
      testCase--;
    }
  }
}
