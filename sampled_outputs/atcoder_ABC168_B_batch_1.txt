import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int k = scanner.nextInt();
        scanner.nextLine();
        String s = scanner.nextLine();
        if (s.length() <= k) {
            System.out.println(s);
        } else {
            s = s.substring(0, k);
            System.out.println(s + "...");
        }
    }
}
