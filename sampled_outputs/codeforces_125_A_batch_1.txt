import java.util.Scanner;

public class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int n = scanner.nextInt();
    int inches = (int) Math.round(n / 3);
    int feet = inches / 12;
    inches = inches % 12;
    System.out.println(feet + " " + inches);
  }
}
