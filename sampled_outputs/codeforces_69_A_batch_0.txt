import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int X = 0, Y = 0, Z = 0;
        for (int i = 0; i < n; i++) {
            int x = sc.nextInt();
            int y = sc.nextInt();
            int z = sc.nextInt();
            X += x;
            Y += y;
            Z += z;
        }
        if (X == Y && Y == Z && Z == 0) {
            System.out.println("YES");
        } else {
            System.out.println("NO");
        }
    }
}
