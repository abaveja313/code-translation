import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int nDays = scanner.nextInt();
        int mBowls = scanner.nextInt();
        int kPlates = scanner.nextInt();
        int washCounter = 0;

        for (int i = 0; i < nDays; i++) {
            String typeDish = scanner.next();
            if (typeDish.equals("1")) {
                if (mBowls == 0) {
                    washCounter++;
                } else {
                    mBowls--;
                }
            } else if (typeDish.equals("2")) {
                if (kPlates > 0) {
                    kPlates--;
                    continue;
                }
                if (mBowls == 0) {
                    washCounter++;
                } else {
                    mBowls--;
                }
            }
        }

        System.out.println(washCounter);
    }
}
