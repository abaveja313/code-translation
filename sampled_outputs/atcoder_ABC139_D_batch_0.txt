import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int b;
        if (n == 1) {
            b = 0;
        } else {
            b = (n * (n - 1)) / 2;
        }
        System.out.println(b);
    }
}
