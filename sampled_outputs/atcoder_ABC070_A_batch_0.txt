public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String n = scanner.nextLine();
        if (n.charAt(0) == n.charAt(n.length() - 1)) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
