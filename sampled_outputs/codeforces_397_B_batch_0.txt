import java.util.Scanner;

public class Solution {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int t = sc.nextInt();
    while (t > 0) {
      t--;
      int n = sc.nextInt();
      int l = sc.nextInt();
      int r = sc.nextInt();
      int k = n / l;
      if (r * k >= n) {
        System.out.println("Yes");
      } else {
        System.out.println("No");
      }
    }
  }
}
