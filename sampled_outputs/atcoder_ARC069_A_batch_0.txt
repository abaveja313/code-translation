import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();
        if (m > n * 2) {
            System.out.println(n + (m - n * 2) / 4);
        } else {
            System.out.println(m / 2);
        }
    }
}
