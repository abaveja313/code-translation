import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int t = sc.nextInt();
        int keepcost = 1000000000;
        for (int i = 0; i < n; i++) {
            int c = sc.nextInt();
            int tt = sc.nextInt();
            if (tt <= t) {
                if (c < keepcost) {
                    keepcost = c;
                }
            }
        }
        System.out.println(keepcost == 1000000000 ? "TLE" : keepcost);
    }
}
