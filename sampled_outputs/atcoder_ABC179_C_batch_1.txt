import java.util.Scanner;

public class Solution {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int N = scanner.nextInt();
    int count = 0;
    for (int i = 1; i < N; i++) {
      count += (N - 1) / i;
    }
    System.out.println(count);
  }
}
