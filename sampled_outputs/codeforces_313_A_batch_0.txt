import java.util.Scanner;

class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();
        if (a > 0) {
            System.out.println(a);
        } else {
            int x = Math.abs(a);
            if (x == 10) {
                System.out.println(0);
            } else {
                int y = x / 10;
                int z = (x / 100) * 10 + x % 10;
                System.out.println(Math.min(y, z) * -1);
            }
        }
    }
}
