import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int N = scanner.nextInt();
        int[] p = new int[N];
        int total = 0;
        for (int i = 0; i < N; i++) {
            p[i] = scanner.nextInt();
        }
        Arrays.sort(p);
        for (int i = 0; i < N; i++) {
            total += p[i];
        }
        System.out.println((int) (total - p[N - 1] / 2));
    }
}
