import java.util.Scanner;

class Solution {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int h = sc.nextInt();
    int[] a = new int[n];
    for (int i = 0; i < n; i++) {
      a[i] = sc.nextInt();
    }
    int count = 0;
    for (int i = 0; i < n; i++) {
      if (a[i] <= h) {
        count++;
      } else {
        count += 2;
      }
    }
    System.out.println(count);
  }
}
